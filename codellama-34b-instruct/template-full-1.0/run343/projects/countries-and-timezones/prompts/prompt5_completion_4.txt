    const tz = data.timezones[id];
    const { c, a } = tz;
    const aliasTz = data.timezones[a] || {};
    const countries = c || aliasTz.c;

    if (!countries) return result;

    countries.forEach((country, i) => {

      if (tz.r === undefined) result[country].current.push(id);
